$(function(){
//	버튼 클릭
	$('#btn1').on('click', function(){
//		전체를 통츨어서 첫번째 li를 선택한다.
		$('li:first').css({
			"background-color" : "yellow"
		});
		
//		ol 그룹마다 첫 번째 자식인 li를 선택한다.
		$('li:first-child').css({
			"border" : "2px inset red"
		});
		
//		ol 그룹마다 짝수 번째 자식인 li를 선택한다.
		$('li:nth-child(even)').css({
			'color' : 'blue'
		});
	});
});
