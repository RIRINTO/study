let rand = parseInt(Math.random() * 100 + 1);
let tryCount = 0;
let findAnswer = false;
function proc1() {
	let number = document.getElementById("num").value;
	let res = "";
	
	if (findAnswer) {
		alert("이미 정답을 맞추셨습니다.")
	} else if (number=="") {
		alert("숫자를 입력해주세요.");
	} else if(isNaN(number)) {
		alert("숫자를 입력해주세요.")
	}else {
		tryCount += 1;
		if (rand == number) {
			res += "입력하신 숫자는 " + number + "입니다. 축하합니다. " + tryCount + "번만에 정답을 맞췄습니다.";
			findAnswer = true;
		} else if (rand < number) {
			res += "Try " + tryCount + ". 입력하신 숫자는 " + number + "입니다. 더 작은 수를 입력하세요.
";
		} else {
			res += "Try " + tryCount + ". 입력하신 숫자는 " + number + "입니다. 더 큰 수를 입력하세요.
";
		}
		document.getElementById("result1").innerHTML += res;
	}
	
}
