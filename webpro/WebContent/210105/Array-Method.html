let arr = ["사과", "복숭아", "참외", "포도", "바나나", "사과", "멜론", "파인애플", "사과"];
arr.push(data);
	: 배열의 마지막 요소로 추가
arr.pop();
	: 배열의 마지막 요소를 꺼내기
arr.reverse();
arr.sort();
arr.slice(start, end);
	: start부터 end - 1까지의 문자열을 잘라내어 새로운 배열을 만든다.ㅇ
		
splice(index, n, a, b, c ···.)
	: 배열을 수정, 삭제
splice(3, 2, 'A', 'B', 'C' ···.)
	: 3번째 위치에서 2개를 삭제하고, 3번째 위치에 'A', 'B', 'C'를 삽입.
splice(2, 0, 'A', 'B', 'C')
	: 2번째 위치에 'A', 'B', 'C'를 삽입.
splice(4, 1)
	: 4번째 위치에서 1개 삭제.
		
배열1.concat(배열2);
	: 여러 개의 배열을 하나의 배열로 합친다.
배열.join(구분자);
	: 하나의 배열을 하나의 문자열로 변환한다.
