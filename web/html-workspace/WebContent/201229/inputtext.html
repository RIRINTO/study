input tag에 id를 기반으로 접근
	id가 in1인 요소에 접근
		var invalue1 = document.getElementById("in1");
	id가 in2인 요소에 접근
		var invalue2 = document.getElementById("in2");
값 가져오기
	: value를 이용
	: 문자열 ( String ) 을 return.
	: parseInt()를 이용하여 정수로 변환
	in1의 값 가져와 v1에 저장
		var v1 = parseInt(invalue1.value);
	in2의 값 가져와 v2에 저장
		var v2 = parseInt(invalue2.value);
input 요소에 결과값 출력
	id가 res1인 요소에 접근
		var res1 = document.getElementById("res1");
	res1의 값을 calcvalue로 변경
		res1.value = calcvalue;
div 요소에 결과값 출력
	id가 res2인 요소에 접근
		var res2 = document.getElementById("res2");
	res2의 텍스트를 calcvalue로 변경
		res2.innerHTML = calcvalue;
