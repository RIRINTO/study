//	객체 정의
	let rect = {
//		속성
//		속성이름 : 속성값,
		width: garo,
		height: sero,
//		함수선언
		getArea: function () {
			return this.width * this.height
		},
		getCircum: function(){
			return (this.width + this.height) * 2
		}
	}
