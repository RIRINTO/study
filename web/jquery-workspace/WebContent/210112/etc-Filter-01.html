$(function(){
	$('#btn1').on('click', function() {

";
		$('#result1 p:hidden').each(function() {
			str += $(this).text() + " ";
		});
		$('#hidden').html(str);
	});
	
	$('#btn2').on('click', function() {
		let str = "
p:visible
";
		$('#result1 p:visible').each(function() {
			str += $(this).text() + " ";
		});
		$('#visible').html(str);
	});
});
