//기존의 이미지를 append 또는 appendTo하면 기존의 이미지가 뒤로 이동한다.
$(function(){
	$('#btn1').on('click', function(){
//		$('#result1 img').first().appendTo($('#result1'));
		$('#result1').append($('#result1 img').first());
	});
});
		
//기존의 이미지를 prepend 또는 prependTo하면 기존의 이미지가 앞으로 이동한다.
$(function(){
	$('#btn2').on('click', function(){
//		$('#result1 img').last().prependTo($('#result1'));
		$('#result2').prepend($('#result2 img').last());
	});
});
