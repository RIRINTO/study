$(function() {
	$('#btn1').on('click', function() {
		let res = "<h3&rt;hidden</h3&rt;";
		$('form *:hidden').each(function() {
//			input tag일 때
			if(this.tagName == "INPUT"){
				res += "input 객체 " + $(this).attr('name') + " " + $(this).val() + "<br>";
			} else {
//			input tag가 아닐 때
				res += this.tagName + " : " + $(this).text() + "<br>";
			}
		});
		$('#hidden').html(res);
	});

	$('#btn2').on('click', function() {
		let res = "<h3>visible</h3>";
		$('form *:visible').each(function() {
//			input tag일 때
			if(this.tagName == "INPUT"){
				/*
				if ($(this).attr('type') == "radio"){
//					type = radio라면 선택된 상태값을 가져온다.
					if($(this).prop('checked')){
						res += $(this).attr('name') + " " + $(this).val() + "<br>";
					}
				} else {
					res += $(this).attr('name') + " " + $(this).val() + "<br>";
				}
				*/
				if ($(this).attr('type') == "radio"){
					if(!$(this).prop('checked')){
						return;
					}
				}
				res += $(this).attr('name') + " " + $(this).val() + "<br>";
			} else {
//			input tag가 아닐 때
				if(this.tagName != "BR"){
					res += this.tagName + " : " + $(this).text();
				}
			}
		});
		$('#visible').html(res);
	});
});
