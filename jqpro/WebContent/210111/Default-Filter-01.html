//준비되면
$(function() {
//	버튼 클릭
	$('#btn1').on('click', function(){
//		첫 번째, 마지막 요소 선택
		$('li:first, li:last').css({
			"background-color" : "yellow"
		});
		
//		4보다 작은 요소 선택
		$('li:lt(4)').css({
			"font-size" : "2.0em",
			"font-weight" : "bold"
		});
		
//		5번째 요소 선택
		$('li:eq(5)').css({
			"background-color" : "red"
		});
	});
});
