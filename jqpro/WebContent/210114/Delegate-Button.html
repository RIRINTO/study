이벤트 등록 방법 - delegate
이미 만들어져 있는 버튼 class=btn과
실행 후에 동적으로 새롭게 만들어진 버튼 class=btn에
같은 이벤트 처리기를 등록 한다.

$('부모 요소').delegate('이벤트 대상 요소', '이벤트 이름', 'function() {
	
});

$('부모 요소').on('이벤트 이름', '이벤트 대상 요소', function() {

});

$('.dd').on('click', '.btn', function(){

});

$(document).on('click', '.btn', function() {

});
